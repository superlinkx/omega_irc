A simple IRC client library compatible with Flutter

Usage

Creating a simple connection:

import 'package:omega_irc/client.dart';

void main() async {
  var host = 'localhost';
  var port = 6667;
  var user = 'derpy';

  var ircClient = Connection(host, port, user);
  ircClient.ircMessage.listen((event) {
    print(event);
  });
}
